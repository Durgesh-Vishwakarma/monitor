BUG 5.6 - HARDWARE NOISE SUPPRESSOR CONFLICT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

File: MicService.kt, Line 3285
Severity: MEDIUM (Audio Quality)
Impact: Prevents dual noise suppression causing poor voice quality

BEFORE:
    if (NoiseSuppressor.isAvailable()) {
        try {
            noiseSuppressor = NoiseSuppressor.create(sid)?.also { e ->
                e.enabled = voiceProfile == "far"  // WRONG: Enables for far mode
            }
        } catch (_: Exception) {}
    }

AFTER:
    if (NoiseSuppressor.isAvailable()) {
        try {
            noiseSuppressor = NoiseSuppressor.create(sid)?.also { e ->
                // Bug 5.6: Disable hardware NS to avoid conflicts with spectral denoiser
                e.enabled = false  // CORRECT: Always disabled
            }
        } catch (_: Exception) {}
    }

REASON:
  - Spectral denoiser is always active in audio capture pipeline
  - Hardware NS conflicts with software spectral denoiser
  - Dual noise suppression causes poor voice quality
  - FIX: Disable hardware NS completely, rely on software spectral denoiser
